дјҳеҠҝпјҡеңЁrun() ж–№жі•еҶ…иҺ·еҸ–еҪ“еүҚзәҝзЁӢзӣҙжҺҘдҪҝз”Ёthis е°ұеҸҜд»ҘдәҶпјҢж— йЎ»дҪҝз”ЁThread.currentThread() ж–№жі•пјӣ

еҠЈеҠҝпјҡжҳҜJava дёҚж”ҜжҢҒеӨҡз»§жүҝпјҢеҰӮжһңз»§жүҝдәҶThread зұ»пјҢйӮЈд№Ҳе°ұдёҚиғҪеҶҚз»§жүҝе…¶д»–зұ»пјӣжІЎжңүзӣёе…іиҝ”еӣһеҖјгҖӮ

дҪҝз”ЁFutureTaskж–№ејҸ

import java.util.concurrent.Callable;
import java.util.concurrent.ExecutionException;
import java.util.concurrent.FutureTask;

public class ThreadTest {
    //з»§жүҝThreadзұ»
    public static class MyThread extends Thread{
        @Override
        public void run(){
            System.out.println("I am a child thread");
        }
    }
	
    //з»§жүҝRunnableжҺҘеҸЈ
    public static class RunnableTask implements Runnable{
        @Override
        public void run() {
            System.out.println("I am a child thread");
        }
    }

    //еҲӣе»әд»»еҠЎпјҢзұ»дјјдәҺRunnable
    public static class CallerTask implements Callable<String> {
        @Override
        public String call() throws Exception{
            return "hello";
        }
    }

    public static void main(String[] args){
        MyThread myThread=new MyThread();
        myThread.start();

        RunnableTask runnableTask=new RunnableTask();
        new Thread(runnableTask).start();
        new Thread(runnableTask).start();
		
        //еҲӣе»әејӮжӯҘд»»еҠЎ
        FutureTask<String> futureTask=new FutureTask<>(new CallerTask());
        //еҗҜеҠЁзәҝзЁӢ
        new Thread(futureTask).start();
        try {
            //зәҝзЁӢжү§иЎҢе®ҢжҲҗиҝ”еӣһз»“жһң
            String result=futureTask.get();
            System.out.println(result);
        }catch (ExecutionException | InterruptedException e){
            e.printStackTrace();
        }
    }
}

2. wait sleepеҢәеҲ«

  1. жҜҸдёӘеҜ№иұЎйғҪжңүдёҖдёӘй”ҒжқҘжҺ§еҲ¶еҗҢжӯҘи®ҝй—®пјҢsynchronizedе…ій”®еӯ—еҸҜд»Ҙе’ҢеҜ№иұЎзҡ„й”ҒдәӨдә’пјҢжқҘе®һзҺ°еҗҢжӯҘж–№жі•жҲ–еҗҢжӯҘеқ—гҖӮsleep()ж–№жі•дјҡи®©жӯЈеңЁжү§иЎҢзҡ„зәҝзЁӢдё»еҠЁи®©еҮәCPUпјҲз„¶еҗҺCPUе°ұеҸҜд»ҘеҺ»жү§иЎҢе…¶д»–д»»еҠЎпјүпјҢеңЁsleepжҢҮе®ҡж—¶й—ҙеҗҺCPUеҶҚеӣһеҲ°иҜҘзәҝзЁӢз»§з»ӯеҫҖдёӢжү§иЎҢ(жіЁж„Ҹпјҡsleepж–№жі•еҸӘи®©еҮәдәҶCPUпјҢиҖҢе№¶дёҚдјҡйҮҠж”ҫеҗҢжӯҘиө„жәҗй”ҒпјҒпјҒпјҒ)пјӣwait()ж–№жі•еҲҷжҳҜжҢҮеҪ“еүҚзәҝзЁӢи®©иҮӘе·ұжҡӮж—¶йҖҖи®©еҮәеҗҢжӯҘиө„жәҗй”ҒпјҢд»Ҙдҫҝе…¶д»–жӯЈеңЁзӯүеҫ…иҜҘиө„жәҗзҡ„зәҝзЁӢеҫ—еҲ°иҜҘиө„жәҗиҝӣиҖҢиҝҗиЎҢпјҢеҸӘжңүи°ғз”ЁдәҶnotify()ж–№жі•пјҢд№ӢеүҚи°ғз”Ёwait()зҡ„зәҝзЁӢжүҚдјҡи§ЈйҷӨwaitзҠ¶жҖҒпјҢеҸҜд»ҘеҺ»еҸӮдёҺз«һдәүеҗҢжӯҘиө„жәҗй”ҒпјҢиҝӣиҖҢеҫ—еҲ°жү§иЎҢгҖӮпјҲжіЁж„Ҹпјҡnotifyзҡ„дҪңз”ЁзӣёеҪ“дәҺеҸ«йҶ’зқЎзқҖзҡ„дәәпјҢиҖҢе№¶дёҚдјҡз»ҷд»–еҲҶй…Қд»»еҠЎпјҢе°ұжҳҜиҜҙnotifyеҸӘжҳҜи®©д№ӢеүҚи°ғз”Ёwaitзҡ„зәҝзЁӢжңүжқғеҲ©йҮҚж–°еҸӮдёҺзәҝзЁӢзҡ„и°ғеәҰпјүпјӣ
  2. sleep()ж–№жі•еҸҜд»ҘеңЁд»»дҪ•ең°ж–№дҪҝз”Ёпјӣwait()ж–№жі•еҲҷеҸӘиғҪеңЁеҗҢжӯҘж–№жі•жҲ–еҗҢжӯҘеқ—дёӯдҪҝз”Ёпјӣ
  3. sleep()жҳҜзәҝзЁӢзәҝзЁӢзұ»пјҲThreadпјүзҡ„ж–№жі•пјҢи°ғз”ЁдјҡжҡӮеҒңжӯӨзәҝзЁӢжҢҮе®ҡзҡ„ж—¶й—ҙпјҢдҪҶзӣ‘жҺ§дҫқз„¶дҝқжҢҒпјҢдёҚдјҡйҮҠж”ҫеҜ№иұЎй”ҒпјҢеҲ°ж—¶й—ҙиҮӘеҠЁжҒўеӨҚпјӣwait()жҳҜObjectзҡ„ж–№жі•пјҢи°ғз”Ёдјҡж”ҫејғеҜ№иұЎй”ҒпјҢиҝӣе…Ҙзӯүеҫ…йҳҹеҲ—пјҢеҫ…и°ғз”Ёnotify()/notifyAll()е”ӨйҶ’жҢҮе®ҡзҡ„зәҝзЁӢжҲ–иҖ…жүҖжңүзәҝзЁӢпјҢжүҚдјҡиҝӣе…Ҙй”Ғжұ пјҢеҶҚж¬ЎиҺ·еҫ—еҜ№иұЎй”ҒжүҚдјҡиҝӣе…ҘиҝҗиЎҢзҠ¶жҖҒ

3.4 еҺҹзҗҶ

Threadзұ»дёӯжңүдёҖдёӘThreadLocalMapеұһжҖ§пјҢThreadLocalMapдёӯEntryзҡ„keyжҳҜThreadLocalеҜ№иұЎпјҢvalueжҳҜObjectзұ»еһӢпјҢдёҖдёӘThreadLocalMapеҸҜд»ҘжңүеӨҡдёӘThreadLocalгҖӮеҰӮеӣҫпјҡThreadзұ»жәҗз ҒдёӯжңүThreadLocalеұһжҖ§гҖӮ

ThreadLocalMap и®ҫи®ЎдёәThreadLocal йқҷжҖҒеҶ…йғЁзұ»гҖӮ

ThreadLocalMap жҳҜдёҖдёӘзәҝзЁӢжң¬ең°зҡ„еҖјпјҢе®ғжүҖжңүзҡ„ж–№жі•йғҪжҳҜprivate зҡ„пјҢд№ҹе°ұж„Ҹе‘ізқҖйҷӨдәҶThreadLocal иҝҷдёӘзұ»пјҢе…¶д»–зұ»жҳҜдёҚиғҪж“ҚдҪңThreadLocalMap дёӯзҡ„д»»дҪ•ж–№жі•зҡ„пјҢиҝҷж ·е°ұеҸҜд»ҘеҜ№е…¶д»–зұ»жҳҜйҖҸжҳҺзҡ„гҖӮеҗҢж—¶иҝҷдёӘзұ»зҡ„жқғйҷҗжҳҜеҢ…зә§еҲ«зҡ„пјҢд№ҹе°ұж„Ҹе‘ізқҖеҸӘжңүеҗҢдёҖдёӘеҢ…дёӢйқўзҡ„зұ»жүҚиғҪеј•з”ЁThreadLocalMap иҝҷдёӘзұ»пјҢиҝҷд№ҹжҳҜThread дёәд»Җд№ҲеҸҜд»Ҙеј•з”ЁThreadLocalMap зҡ„еҺҹеӣ пјҢеӣ дёәд»–д»¬еңЁеҗҢдёҖдёӘеҢ…дёӢйқўгҖӮ

иҷҪз„¶Thread еҸҜд»Ҙеј•з”ЁThreadLocalMapпјҢдҪҶжҳҜдёҚиғҪи°ғз”Ёд»»дҪ•ThreadLocalMap дёӯзҡ„ж–№жі•гҖӮиҝҷд№ҹе°ұжҳҜжҲ‘д»¬е№іж—¶йғҪжҳҜйҖҡиҝҮThreadLocal жқҘиҺ·еҸ–еҖје’Ңи®ҫзҪ®еҖјгҖӮ

ThreadLocal setж–№жі•

  1. иҺ·еҸ–еҪ“еүҚзәҝзЁӢпјҢж №жҚ®еҪ“еүҚзәҝзЁӢиҺ·еҸ–еҜ№еә”зҡ„ThreadLocalMap
  2. еҰӮжһңеҜ№еә”зҡ„ThreadLocalMapдёҚдёәnullпјҢеҲҷи°ғз”Ёе…¶зҡ„setж–№жі•дҝқеӯҳеҜ№еә”е…ізі»
  3. еҰӮжһңmapдёәnullпјҢе°ұжңҖз»Ҳи°ғз”ЁThreadLocalMapзҡ„жһ„йҖ ж–№жі•еҲӣе»әдёҖдёӘThreadLocalMapе№¶дҝқеӯҳеҜ№еә”е…ізі»
public void set(T value) {
    Thread t = Thread.currentThread();
    ThreadLocalMap map = getMap(t);
    if (map != null)
        map.set(this, value);
    else
        createMap(t, value);
}
ThreadLocalMap getMap(Thread t) {
    return t.threadLocals;
}

ThreadLocal getж–№жі•

  1. иҺ·еҸ–еҪ“еүҚзәҝзЁӢпјҢж №жҚ®еҪ“еүҚзәҝзЁӢиҺ·еҸ–еҜ№еә”зҡ„ThreadLocalMap
  2. еңЁThreadLocalMapеҪ“дёӯиҺ·еҸ–иҜҘThreadLocalеҜ№иұЎеҜ№еә”зҡ„EntryиҠӮзӮ№пјҢе№¶дё”иҝ”еӣһеҜ№еә”зҡ„еҖј
  3. еҰӮжһңиҺ·еҸ–еҲ°зҡ„ThreadLocalMapдёәnullпјҢеҲҷиҜҒжҳҺиҝҳжІЎжңүеҲқе§ӢеҢ–пјҢе°ұи°ғз”ЁsetInitialValueж–№жі•
    • еңЁи°ғз”ЁsetInitialValueж–№жі•зҡ„ж—¶еҖҷпјҢдјҡеҸҢйҮҚдҝқиҜҒпјҢеҶҚиҝӣиЎҢиҺ·еҸ–дёҖж¬ЎThreadLocalMap
    • еҰӮжһңдҫқз„¶дёәnullпјҢе°ұжңҖз»Ҳи°ғз”ЁThreadLocalMapзҡ„жһ„йҖ ж–№жі•
public T get() {
       //иҝҷйҮҢйҖҡиҝҮиҺ·еҸ–еҪ“еүҚзҡ„зәҝзЁӢ
       Thread t = Thread.currentThread();
       //йҖҡиҝҮзәҝзЁӢжқҘиҺ·еҸ–ThreadLocalMap пјҢиҝҳи®°еҫ—жҲ‘д»¬дёҠйқўиҜҙзҡ„Thread йҮҢйқўжңүдёҖдёӘThreadLocalMap еұһжҖ§еҗ—пјҹе°ұжҳҜиҝҷйҮҢз”ЁдёҠдәҶ
       ThreadLocalMap map = getMap(t);
       if (map != null) {
           ThreadLocalMap.Entry e = map.getEntry(this);
           if (e != null) {
               @SuppressWarnings("unchecked")
               T result = (T)e.value;
               return result;
           }
       }
       return setInitialValue();
   }
ThreadLocalMap getMap(Thread t) {
    return t.threadLocals;
}

3.5 еҜ№иұЎеӯҳж”ҫеңЁе“ӘйҮҢ

еңЁJavaдёӯпјҢж ҲеҶ…еӯҳеҪ’еұһдәҺеҚ•дёӘзәҝзЁӢпјҢжҜҸдёӘзәҝзЁӢйғҪдјҡжңүдёҖдёӘж ҲеҶ…еӯҳпјҢе…¶еӯҳеӮЁзҡ„еҸҳйҮҸеҸӘиғҪеңЁе…¶жүҖеұһзәҝзЁӢдёӯеҸҜи§ҒпјҢеҚіж ҲеҶ…еӯҳеҸҜд»ҘзҗҶи§ЈжҲҗзәҝзЁӢзҡ„з§ҒжңүеҶ…еӯҳгҖӮиҖҢе ҶеҶ…еӯҳдёӯзҡ„еҜ№иұЎеҜ№жүҖжңүзәҝзЁӢеҸҜи§ҒгҖӮе ҶеҶ…еӯҳдёӯзҡ„еҜ№иұЎеҸҜд»Ҙиў«жүҖжңүзәҝзЁӢи®ҝй—®гҖӮ

йӮЈд№ҲжҳҜдёҚжҳҜиҜҙThreadLocalзҡ„е®һдҫӢд»ҘеҸҠе…¶еҖјеӯҳж”ҫеңЁж ҲдёҠе‘ўпјҹ

е…¶е®һдёҚжҳҜпјҢеӣ дёәThreadLocalе®һдҫӢе®һйҷ…дёҠд№ҹжҳҜиў«е…¶еҲӣе»әзҡ„зұ»жҢҒжңүпјҲжӣҙйЎ¶з«Ҝеә”иҜҘжҳҜиў«зәҝзЁӢжҢҒжңүпјүгҖӮиҖҢThreadLocalзҡ„еҖје…¶е®һд№ҹжҳҜиў«зәҝзЁӢе®һдҫӢжҢҒжңүгҖӮ

е®ғд»¬йғҪжҳҜдҪҚдәҺе ҶдёҠпјҢеҸӘжҳҜйҖҡиҝҮдёҖдәӣжҠҖе·§е°ҶеҸҜи§ҒжҖ§дҝ®ж”№жҲҗдәҶзәҝзЁӢеҸҜи§ҒгҖӮ

3.6 йҖҡиҝҮinheritableThreadLocalе®һзҺ°еӨҡдёӘзәҝзЁӢи®ҝй—®ThreadLocalзҡ„еҖј

дҪҝз”ЁInheritableThreadLocalеҸҜд»Ҙе®һзҺ°еӨҡдёӘзәҝзЁӢи®ҝй—®ThreadLocalзҡ„еҖјгҖӮ

еҰӮдёӢпјҢжҲ‘д»¬еңЁдё»зәҝзЁӢдёӯеҲӣе»әдёҖдёӘInheritableThreadLocalзҡ„е®һдҫӢпјҢз„¶еҗҺеңЁеӯҗзәҝзЁӢдёӯеҫ—еҲ°иҝҷдёӘInheritableThreadLocalе®һдҫӢи®ҫзҪ®зҡ„еҖјгҖӮ

3.7 дҪҝз”ЁеңәжҷҜ

  • е®һзҺ°еҚ•дёӘзәҝзЁӢеҚ•дҫӢд»ҘеҸҠеҚ•дёӘзәҝзЁӢдёҠдёӢж–ҮдҝЎжҒҜеӯҳеӮЁпјҢжҜ”еҰӮдәӨжҳ“idзӯү
  • е®һзҺ°зәҝзЁӢе®үе…ЁпјҢйқһзәҝзЁӢе®үе…Ёзҡ„еҜ№иұЎдҪҝз”ЁThreadLocalд№ӢеҗҺе°ұдјҡеҸҳеҫ—зәҝзЁӢе®үе…ЁпјҢеӣ дёәжҜҸдёӘзәҝзЁӢйғҪдјҡжңүдёҖдёӘеҜ№еә”зҡ„е®һдҫӢ
  • жүҝиҪҪдёҖдәӣзәҝзЁӢзӣёе…ізҡ„ж•°жҚ®пјҢйҒҝе…ҚеңЁж–№жі•дёӯжқҘеӣһдј йҖ’еҸӮж•°
